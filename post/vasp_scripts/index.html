2.截断能测试

rm summary
for i in 250 300 350 400 450 500 550 600 650
do
cat>INCAR<<!
SYSTEM=Al13
NSW=0
ENCUT=$i
ISMEAR=0
EDIFFG=-0.01
LREAL=Auto
EDIFF=1e-06
NPAR=2
ISIF=2
ISPIN=2
IBRION=-1

!

mpirun -n 12 vasp_std
E=`tail -1 OSZICAR`
inf=`echo $E|cut -d " " -f 1,3,5,10`
echo $i $inf >>summary
mv OSZICAR OSZICAR-$i
mv CONTCAR CONTCAR-$i
rm WAVECAR DOSCAR
done
